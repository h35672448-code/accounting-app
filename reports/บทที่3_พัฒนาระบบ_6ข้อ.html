function doGet() {
  const page = "index";
  const output = HtmlService.createHtmlOutputFromFile(page);
  output.setTitle("ทะเบียนคุม");
  output.addMetaTag("viewport", "width=device-width, initial-scale=1");
  return output;
}

ฟังก์ชันนี้เป็นจุดเริ่มต้นของระบบ เมื่อผู้ใช้เปิดลิงก์เว็บแอป ระบบจะส่งหน้า index กลับไปแสดงผลทันที พร้อมตั้งค่าหน้าจอให้รองรับมือถือ.

ข้อที่ 2 การตรวจสอบสิทธิ์ผู้ใช้งาน

function login(password) {
  const pw = String(password || "").trim();
  if (pw === "0000") return "admin";
  if (pw === "1111") return "user";
  return "";
}

ฟังก์ชันนี้รับรหัสผ่านจากผู้ใช้ แล้วกำหนดบทบาทเป็น admin หรือ user. หากรหัสไม่ถูกต้องจะคืนค่าว่างเพื่อป้องกันการเข้าระบบ.

ข้อที่ 3 การเพิ่มข้อมูลใหม่

function addRow(row) {
  const valid = validateAndCheckDuplicate(row, null);
  if (!valid.ok) throw new Error("ข้อมูลไม่ผ่านการตรวจสอบ");
  const rows = readAll();
  rows.push(row);
  saveAll(rows);
  appendAudit("ADD", { row: row });
}

ส่วนนี้ใช้บันทึกรายการบัญชีใหม่ โดยตรวจความถูกต้องก่อนเสมอ จากนั้นจึงบันทึกข้อมูลและเก็บประวัติการทำรายการ (Audit Log).

ข้อที่ 4 การตรวจความถูกต้องและข้อมูลซ้ำ

function validateAndCheckDuplicate(row, editIndex){
  const errors = [];
  const date = String((row && row[0]) || "").trim();
  const doc = String((row && row[1]) || "").trim();
  if (!date) errors.push("วันที่ห้ามว่าง");
  if (!doc) errors.push("เลขที่เอกสารห้ามว่าง");
  const dup = findDuplicateRow(row, editIndex);
  return { ok: errors.length === 0, errors: errors, duplicate: !!dup, duplicateRow: dup };
}

ฟังก์ชันนี้ช่วยลดข้อผิดพลาดจากการบันทึกข้อมูล โดยตรวจช่องสำคัญที่ห้ามว่าง และตรวจว่ามีรายการซ้ำหรือไม่ก่อนยืนยันการบันทึก.

ข้อที่ 5 การนำเข้าข้อมูลจากไฟล์

function importData(obj){
  const name = obj && obj.name ? obj.name.toLowerCase() : "";
  if (!name) throw new Error("ไม่มีข้อมูลไฟล์");
  if (name.endsWith(".csv")) { replaceCSV(obj.data); return "ok"; }
  if (name.endsWith(".xlsx")) { replaceXLSX(obj.data, obj.encoding); return "ok"; }
  throw new Error("รองรับเฉพาะ .csv หรือ .xlsx");
}

โค้ดส่วนนี้รองรับการนำเข้าทั้ง CSV และ Excel โดยตรวจชนิดไฟล์ก่อนนำเข้า เพื่อป้องกันไฟล์ผิดรูปแบบและลดปัญหาข้อมูลพัง.

ข้อที่ 6 การส่งออกข้อมูลที่เลือกเป็น Excel

function downloadSelected(indexes) {
  const all = readAll();
  const header = all[0];
  const rows = (indexes || []).map(function(i){ return all[i]; }).filter(Boolean);
  const dataRows = [header].concat(rows);
  const xlsxBlob = buildXlsx(dataRows);
  return { name: "ทะเบียนคุม_ส่งออก.xlsx", type: xlsxBlob.getContentType(), data: Utilities.base64Encode(xlsxBlob.getBytes()) };
}

ฟังก์ชันนี้ใช้ส่งออกเฉพาะแถวที่ผู้ใช้เลือก ช่วยให้ได้ไฟล์รายงานตรงความต้องการและลดขั้นตอนคัดข้อมูลภายหลัง.
