ระบบ: ทะเบียนคุมบันทึกการปรับปรุงรายการบัญชี (Google Apps Script + HTML/JS)

ข้อ 1: จุดเริ่มต้นเว็บแอป (doGet)

function doGet() {
  return HtmlService
    .createHtmlOutputFromFile("index");
}

อธิบาย: ฟังก์ชันนี้เป็นประตูหลักของระบบ เมื่อผู้ใช้เปิดลิงก์ Web App ระบบจะโหลดหน้า index.html เพื่อแสดงฟอร์มล็อกอินและหน้าจอใช้งานทั้งหมด

ข้อ 2: ระบบเข้าสู่ระบบและกำหนดสิทธิ์ (login)

function login(password) {
  const pw = String(password || "").trim();
  if (pw === "0000") return "admin";
  if (pw === "1111") return "user";
  return "";
}

อธิบาย: ตรวจรหัสผ่านและส่งบทบาทผู้ใช้กลับไปหน้าเว็บ โดย admin จะมีสิทธิ์เพิ่ม/แก้ไข/ลบ ส่วน user เน้นการบันทึกและดูข้อมูลตามสิทธิ์ที่กำหนด

ข้อ 3: การบันทึกข้อมูลรายการบัญชี (addRow)

function addRow(row) {
  const valid = validateAndCheckDuplicate(row, null);
  if (!valid.ok) throw new Error("ข้อมูลไม่ผ่านการตรวจสอบ");
  const rows = readAll();
  rows.push(row);
  saveAll(rows);
  appendAudit("ADD", { row: row });
}

อธิบาย: ก่อนบันทึกจริง ระบบจะตรวจความถูกต้องของข้อมูล (วันที่, เอกสาร, จำนวนเงิน) และตรวจรายการซ้ำ จากนั้นจึงบันทึกลงไฟล์ พร้อมเขียน Audit Log เพื่อใช้ตรวจสอบย้อนหลัง

ข้อ 4: นำเข้าข้อมูลจากไฟล์ (importData)

function importData(obj){
  const name = (obj.name || "").toLowerCase();
  if(name.endsWith(".csv")){ replaceCSV(obj.data); return "ok"; }
  if(name.endsWith(".xlsx")){ replaceXLSX(obj.data, obj.encoding); return "ok"; }
  throw new Error("รองรับเฉพาะ .csv หรือ .xlsx");
}

อธิบาย: ระบบรองรับการนำเข้าทั้ง CSV และ Excel (.xlsx) โดยตรวจชนิดไฟล์ก่อน หากไม่ตรงรูปแบบจะหยุดและแจ้งข้อผิดพลาดทันที เพื่อลดข้อมูลเสียในระบบ

ข้อ 5: ตรวจโครงสร้างหัวตารางก่อนรวมข้อมูล (validateHeader)

function validateHeader(fileHeader){
  const cleaned = fileHeader.map(normalizeHeaderText);
  const expected = HEADER.map(normalizeHeaderText);
  for (let i=0; i<expected.length; i++) {
    if (!headerCellMatches(cleaned[i], expected[i])) {
      throw new Error("หัวตารางไม่ตรงคอลัมน์ " + (i+1));
    }
  }
}

อธิบาย: ก่อนนำเข้าข้อมูล ระบบจะเทียบหัวตารางกับมาตรฐานที่กำหนดไว้ (วันเดือนปี, เลขที่เอกสาร, รายการ, RE/JR/JV/PP ฯลฯ) เพื่อป้องกันคอลัมน์สลับตำแหน่ง

ข้อ 6: ส่งออกข้อมูลที่เลือกเป็นไฟล์ Excel (downloadSelected)

function downloadSelected(indexes) {
  const all = readAll();
  const header = all[0];
  const rows = (indexes || []).map(i => all[i]).filter(Boolean);
  const xlsxBlob = buildXlsx([header].concat(rows));
  return {
    name: "ทะเบียนคุม_ส่งออก.xlsx", data: Utilities.base64Encode(xlsxBlob.getBytes())
  };
}
